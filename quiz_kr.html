    <h1>효과충 vs 효율충 테스트</h1>

    <div id="quiz"></div>

  </div>


  <script>

    const questions = [

      {

        q: '일이 시작되었을 때, 나는?',

        a: '일단 전체 목적과 방향을 먼저 파악한다.',

        b: '일단 빠르게 처리 가능한 것부터 손댄다.',

      },

      {

        q: '새로운 프로젝트를 맡았을 때, 나는?',

        a: '이 일이 왜 필요한지, 어떤 의미인지 이해해야 동기부여 된다.',

        b: '구체적인 액션 플랜과 마일스톤이 있어야 안정감을 느낀다.',

      },

      // ...이하 8문항 추가 가능 (총 10문항)

    ];


    let current = 0;

    let scoreA = 0;

    let scoreB = 0;


    const quizEl = document.getElementById('quiz');


    function renderQuestion() {

      if (current >= questions.length) {

        showResult();

        return;

      }


      const q = questions[current];

      quizEl.innerHTML = `

        <div class="question">${q.q}</div>

        <button class="btn" onclick="choose('a')">${q.a}</button>

        <button class="btn" onclick="choose('b')">${q.b}</button>

      `;

    }


    function choose(answer) {

      if (answer === 'a') scoreA++;

      if (answer === 'b') scoreB++;

      current++;

      renderQuestion();

    }


    function showResult() {

      let result = '';

      if (scoreA >= 7) {

        result = '💡 당신은 효과충입니다! Meaning-Oriented Type';

      } else if (scoreB >= 7) {

        result = '⚙️ 당신은 효율충입니다! Efficiency-Seeker Type';

      } else {

        result = '⚖️ 당신은 밸런서입니다! Strategy-in-Action Type';

      }


      quizEl.innerHTML = `<h2>${result}</h2>`;

    }


    renderQuestion();

  </script>
